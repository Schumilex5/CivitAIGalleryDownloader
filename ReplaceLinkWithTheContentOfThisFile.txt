javascript:(async()=>{const imgs=Array.from(document.querySelectorAll('img[src*="image.civitai.com"]'));if(!imgs.length){alert('No images found. Scroll through the gallery first so images load.');return;}const urls=imgs.map(img=>img.src.replace(/\/anim=.*?\/|,optimized=true|,width=\d+/g,'/').split('?')[0]).filter(u=>u&&u.startsWith('https'));console.log(`Found ${urls.length} images. Fetching and saving...`);for(let i=0;i<urls.length;i++){const url=urls[i];const filename=`civitai_image_${i+1}.jpg`;try{const res=await fetch(url);const blob=await res.blob();const blobUrl=URL.createObjectURL(blob);const a=document.createElement('a');a.href=blobUrl;a.download=filename;document.body.appendChild(a);a.click();a.remove();URL.revokeObjectURL(blobUrl);console.log(`✅ Saved: ${filename}`);}catch(err){console.error(`❌ Failed to fetch ${url}`,err);}}alert(`All ${urls.length} images processed! Check your Downloads folder.`);})();
